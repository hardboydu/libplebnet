BIND Configuration File Guide -- Example Config File


/*
 * A simple BIND 8 configuration
 */

options {
	directory "/var/named";
};

logging {
	category lame-servers { null; };
	category cname { null; };
};

zone "isc.org" in {
	type master;
	file "master/isc.org";
};

zone "vix.com" in {
	type slave;
	file "slave/vix.com";
	masters { 10.0.0.53; };
};

zone "." in {
	type hint;
	file "named.cache";
};

zone "0.0.127.in-addr.arpa" in {
	type master;
	file "master/127.0.0";
};
